FIX #1: DreamV3DdiDpcRoutine - Infinite Loop Protection
--------

LOCATION: amdbc250_dream_v3_dpc.c (Interrupt handler DPC routine)

BEFORE (BUGGY):
```c
void DreamV3DdiDpcRoutine(PVOID MiniportDeviceContext)
{
    PDREAM_V3_DEVICE_EXTENSION DevExt = (PDREAM_V3_DEVICE_EXTENSION)MiniportDeviceContext;
    ULONG RPtr = DreamV3ReadRegister(DevExt, AMDBC250_REG_IH_RB_RPTR);
    ULONG WPtr = DreamV3ReadRegister(DevExt, AMDBC250_REG_IH_RB_WPTR);
    
    while (RPtr != WPtr) {  // ❌ INFINITE LOOP if WPtr keeps advancing
        // Process interrupt entry...
        RPtr += IH_ENTRY_SIZE_BYTES;
        if (RPtr >= DevExt->IhRing.SizeInBytes) {
            RPtr = 0;
        }
    }
}
```

AFTER (FIXED):
```c
void DreamV3DdiDpcRoutine(PVOID MiniportDeviceContext)
{
    PDREAM_V3_DEVICE_EXTENSION DevExt = (PDREAM_V3_DEVICE_EXTENSION)MiniportDeviceContext;
    ULONG RPtr = DreamV3ReadRegister(DevExt, AMDBC250_REG_IH_RB_RPTR);
    ULONG WPtr = DreamV3ReadRegister(DevExt, AMDBC250_REG_IH_RB_WPTR);
    
    // ✅ FIXED: Add safety limit to prevent infinite loops
    ULONG MaxEntries = DevExt->IhRing.SizeInBytes / IH_ENTRY_SIZE_BYTES;
    ULONG ProcessedCount = 0;
    
    while (RPtr != WPtr && ProcessedCount < MaxEntries) {
        // Process interrupt entry...
        ProcessedCount++;
        RPtr += IH_ENTRY_SIZE_BYTES;
        if (RPtr >= DevExt->IhRing.SizeInBytes) {
            RPtr = 0;
        }
    }
    
    // If we hit the limit, log warning
    if (ProcessedCount >= MaxEntries) {
        DbgPrintEx(DPFLTR_IHVDRIVER_ID, DPFLTR_ERROR_LEVEL,
                   "BC-250 KMD: DPC processed max entries (%u), possible loop condition\n",
                   MaxEntries);
    }
    
    // Update read pointer
    DreamV3WriteRegister(DevExt, AMDBC250_REG_IH_RB_RPTR, RPtr);
}
```

FIX #2: DreamV3DdiCreateAllocation - Memory Leak Prevention
-----------

LOCATION: amdbc250_dream_v3_alloc.c (Allocation creation)

BEFORE (BUGGY):
```c
NTSTATUS DreamV3DdiCreateAllocation(
    CONST HANDLE hAdapter,
    DXGKARG_CREATEALLOCATION *pCreateAllocation)
{
    PDREAM_V3_DEVICE_EXTENSION DevExt = ...;
    
    for (UINT i = 0; i < pCreateAllocation->NumAllocations; i++) {
        PDREAM_V3_ALLOCATION pAlloc = ExAllocatePoolWithTag(...);  // ❌ No cleanup if fails
        
        ExAcquireFastMutex(&DevExt->DeviceMutex);  // ❌ Acquires BEFORE alloc check
        InsertTailList(&DevExt->AllocationList, &Alloc->ListEntry);
        ExReleaseFastMutex(&DevExt->DeviceMutex);
        
        // If next allocation fails, already inserted allocations leaked
    }
}
```

AFTER (FIXED):
```c
NTSTATUS DreamV3DdiCreateAllocation(
    CONST HANDLE hAdapter,
    DXGKARG_CREATEALLOCATION *pCreateAllocation)
{
    PDREAM_V3_DEVICE_EXTENSION DevExt = ...;
    PDREAM_V3_ALLOCATION* ppAllocations = NULL;
    
    // ✅ FIXED: Pre-allocate array of pointers for all allocations
    ppAllocations = ExAllocatePoolWithTag(NonPagedPool, 
        pCreateAllocation->NumAllocations * sizeof(PDREAM_V3_ALLOCATION),
        DREAM_V3_TAG_ALLOCATION);
    
    if (!ppAllocations) {
        return STATUS_NO_MEMORY;
    }
    
    RtlZeroMemory(ppAllocations, 
        pCreateAllocation->NumAllocations * sizeof(PDREAM_V3_ALLOCATION));
    
    // ✅ FIXED: Allocate all first, then insert (atomic operation)
    for (UINT i = 0; i < pCreateAllocation->NumAllocations; i++) {
        PDREAM_V3_ALLOCATION pAlloc = ExAllocatePoolWithTag(...);
        
        if (!pAlloc) {
            // ✅ FIXED: Cleanup previously allocated items
            for (UINT j = 0; j < i; j++) {
                ExFreePoolWithTag(ppAllocations[j], DREAM_V3_TAG_ALLOCATION);
            }
            ExFreePoolWithTag(ppAllocations, DREAM_V3_TAG_ALLOCATION);
            return STATUS_NO_MEMORY;
        }
        
        ppAllocations[i] = pAlloc;
    }
    
    // ✅ FIXED: Only acquire mutex once, insert all at once
    ExAcquireFastMutex(&DevExt->DeviceMutex);
    
    for (UINT i = 0; i < pCreateAllocation->NumAllocations; i++) {
        InsertTailList(&DevExt->AllocationList, &ppAllocations[i]->ListEntry);
        DevExt->SubmitCount++;  // Track for statistics
    }
    
    ExReleaseFastMutex(&DevExt->DeviceMutex);
    
    ExFreePoolWithTag(ppAllocations, DREAM_V3_TAG_ALLOCATION);
    
    return STATUS_SUCCESS;
}
```
